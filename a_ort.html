      Im Laufe der nächsten Übungen werden wir uns fÜr das Damenproblem (Englisch: "Eight queens puzzle") interessieren. 
      In dem Problem geht es darum, auf einem `n × n`-Brett genau `n` Damen zu platzieren, ohne dass die eine die andere angreifen kann. 
      Zwei Damen dÜrfen also niemals auf derselben Diagonale oder Linie sein. 
      Das Problem besteht darin herauszufinden, wie viele Lösungen es gibt. Lösungen können
      Symmetrien aufweisen, aber das interessiert uns erst einmal nicht.
      In dem Folgende werden wir dieses Problem mit einer besonderen Darstellung lösen: BDDs (binary decision diagrams). 
      Das ist ein Graph (im Moment: ein Baum) von Knoten. 
      Jeder Knoten beschreibt eine Position einer Dame und stellt die Frage ¨Ist die Dame an der nächsten Stelle da?”. 
      Jeder Knoten hat dann zwei Nachfolger : “wenn sie da ist” und “wenn sie nicht da ist”, daher auch der Name “If-Then-Else” fÜr jeden Knoten.
      Es gibt zwei besondere Blätter: 0 (keine Lösung zum Damenproblem) und 1 (Lösung zum Damenproblem).
      FÜr den Nachfolger einer Position stellen wir eine Dame in der ersten leeren Spalte (von links
      nach rechts) und setzen sie dort ein oder setzen keine Dame an dieser Stelle.
      Eine Position ist:
       *voll* (full) wenn alle Damen auf dem Brett stehen;
      *(in)korrekt* (valid), wenn (nicht) alle Regeln befolgt werden (also dass zwei Damen sind nehmen können)
      Achtung Das leere Brett ist korrekt, aber nicht voll und deshalb auch keine Lösung zum Damenproblem.
      
      Auf dem `bild01.png` ist der Anfang von einem BDD gezeichnet. In jedem Knoten ist ein 4 × 4 Brett, bei dem d bedeutet, dass eine Dame dort ist, . bedeutet, dass dort keine Dame stehen kann. Die
      vollen Pfeile bedeuten “Wenn die Dame dort ist” und die anderen “Wenn die Dame nicht dort
      ist”.
      Einer der Algorithmen, den wir hier implementieren wollen, ist folgender:
      void baue_den_BDD_auf(die Größe n)
      Stappel positionen_zu_bearbeiten := [keine_Damen] [Aufgabe 1]

      for beide Nachfolger "ist da" und "ist nicht da" von der Position P
      wenn der Nachfolger voll ist oder [Aufgabe 1]
      der Nachfolger inkorrekt ist [Aufgabe 1]
      setze 0 oder 1 als Nachfolger im BDD [üB 8, Aufgabe 4]
      sonst
      füge den Nachfolger im BDD und in positionen_zu_bearbeiten

      Node *BDD::import_node(const Board &b, bool &already_exists) {
        Board a = b;  // Wir kopieren das Brett, um es zu normalisieren, falls nötig.

            a.has_symmetries();  // Wir prüfen, ob das Brett Symmetrien hat
      
        Node *n;
      
        if (not_normalized) {  // Wenn das Brett Symmetrien hat, normalisieren wir es.

                            // normalisierten Brett.

      int BDD::count_true_path() const {
        // We create a stack for storing the nodes to visit
        Stack stack;
        // We create a variable for storing the number of paths
        int path = 0;
        // We check if the root node is null
        if (!root) return 0;
        // We push the root node into the stack
        stack.push(root);
        // We repeat until the stack is empty
        while (!stack.empty()) {
          // We pop the top node from the stack
          Node *n = stack.top();
          stack.pop();
          // We check if the node is true
          if (n == true_node) {
            // We increment the path variable
            ++path;
            // We skip the rest of the loop
            continue;
          }
          // We check if the node is false
          if (n == false_node) {
            // We skip the rest of the loop
            continue;
          }
          // We push both successors of the node into the stack
          stack.push(n->if_true);
          stack.push(n->if_false);
        }
        // We return the path variable
        return path;
      }
      
      int BDD::count_fundamental_true_path() const {
        // Diese Funktion soll die Anzahl der fundamentalen Lösungen zählen, indem sie
        // jeden Knoten nur einmal besucht
      
        // Wir benutzen eine rekursive Hilfsfunktion, die einen Knoten als Parameter
        // nimmt und die Anzahl der fundamentalen Lösungen von diesem Knoten aus
        // zurückgibt.
      
        // Wir definieren die Hilfsfunktion als eine Lambda-Funktion innerhalb der
        // Funktion.
      
        std::function count_helper = [&](Node *n) -> int {
          if (n == true_node)
            return 1;  // Wenn der Knoten true ist, gibt es eine fundamentale Lösung.
      
          if (n == false_node)
            return 0;  // Wenn der Knoten false ist, gibt es keine fundamentale
                       // Lösung.
      
          if (n->to_delete)
            return 0;  // Wenn der Knoten schon besucht wurde, geben wir 0 zurück, um
                       // ihn nicht doppelt zu zählen.
      
          n->to_delete = true;  // Wir markieren den Knoten als besucht, indem wir den
                                // to_delete-Flag auf true setzen.
      
          // Wir rufen die Hilfsfunktion rekursiv für beide Nachfolger auf und
          // addieren die Ergebnisse.
      
          int result = count_helper(n->if_true) + count_helper(n->if_false);
      
          n->to_delete = false;  // Wir setzen den to_delete-Flag wieder auf false, um
                                 // den Zustand des BDD nicht zu verändern.
      
          return result;  // Wir geben das Ergebnis zurück.
        };
      
        // Wir rufen die Hilfsfunktion für den Wurzelknoten auf und geben das Ergebnis
        // zurück.
      
        return count_helper(root);
      }

      // We implement a recursive post-order traversal of the BDD
      bool BDD::is_removable(Node *b) {
        /*NEW ÜB 10, Aufgabe 11*/
        // Diese Funktion soll überprüfen, ob ein Knoten entfernt werden kann, indem
        // er alle Pfade zu false hat. Wir gehen davon aus, dass b kein Blatt ist
        // (weder true noch false).
        assert(b != true_node && b != false_node);
      
        // Wir prüfen zuerst, ob die Nachfolger entfernt werden können, indem wir die
        // Funktion rekursiv aufrufen.
        bool if_true_removable = is_removable(b->if_true);
        bool if_false_removable = is_removable(b->if_false);
      
        // Wenn ein Nachfolger entfernt werden kann, ersetzen wir ihn durch den
        // false-Knoten im BDD.
      if (if_true_removable) {
        this->connect_false(b, true); // oder einfach connect_false(b, true);
      }
      if (if_false_removable) {
        this->connect_false(b, false); // oder einfach connect_false(b, false);
      }
      
        // Jetzt prüfen wir, ob der aktuelle Knoten entfernt werden kann.
        // Das ist der Fall, wenn beide Nachfolger false sind.
        if (b->if_true == false_node && b->if_false == false_node) {
          // Wir markieren den Knoten als entfernt, indem wir den to_delete-Flag auf
          // true setzen.
          b->to_delete = true;
          // Wir geben true zurück, um anzuzeigen, dass der Knoten entfernt werden
          // kann.
          return true;
        }
      
        // Ansonsten geben wir false zurück, um anzuzeigen, dass der Knoten nicht
        // entfernt werden kann.
        return false;
      }
      
      // remove useless parts of the entire BDD, by checking if the root should be
      // removed
      void BDD::normalize_bdd() {
        /*NEW ÜB 10, Aufgabe 11*/
        // Diese Funktion soll den gesamten BDD reduzieren, indem sie alle Knoten entfernt, die nur Pfade zu false haben.
        // Wir prüfen zuerst, ob der Wurzelknoten entfernt werden kann, indem wir die Funktion is_removable aufrufen.
        bool root_removable = is_removable(root);
        
        // Wenn der Wurzelknoten entfernt werden kann, setzen wir ihn auf den false-Knoten.
        if (root_removable) {
          root = false_node;
          return; // Wir sind fertig, da es keine anderen Knoten mehr gibt.
        }
      
        // Ansonsten iterieren wir über alle Knoten im BDD und löschen diejenigen, die als entfernt markiert sind.
        for (auto it = nodes.begin(); it != nodes.end(); ) {
          Node *n = *it;
          if (n->to_delete) { // Wenn der to_delete-Flag auf true gesetzt ist, löschen wir den Knoten aus dem std::set und aus dem Speicher.
            it = nodes.erase(it); // Die erase-Methode gibt einen Iterator auf das nächste Element zurück.
            delete n; // Wir geben den Speicherplatz frei.
          } else { // Wenn der to_delete-Flag auf false gesetzt ist, gehen wir zum nächsten Element weiter.
            ++it;
          }
        }
      }

      bool Board::has_next(bool taken) const {
        int position_to_change = -1;
        for (int i = 0; i < size; ++i) {
          if (board[i] <= 0) {
            position_to_change = i;
            break;
          }
        }
      
        if (position_to_change == -1) return false;
      
        const int value = board[position_to_change];
      
        if (!taken && size == -value + 1) return false;
      
        return true;
      }
      
      void Board::next(bool taken) {
        int position_to_change = -1;
        for (int i = 0; i < size; ++i) {
          if (board[i] <= 0) {
            position_to_change = i;
            break;
          }
        }
      
        assert(position_to_change != -1);
      
        const int value = board[position_to_change];
      
        assert(taken || size != value - 1);
      
        if (taken)
          board[position_to_change] = -value + 1;
        else
          board[position_to_change] -= 1;
      }

      // Checks if a position is valid: two queens are (i) on the same row if they
      // have the position and (ii) are on the same diagonal if the distance is the
      // same as the row distance.
      bool Board::is_valid() const {
        for (int i = 0; i < size; ++i) {
          if (board[i] < 0) continue;
          if (board[i] > 0) {
            for (int j = 0; j < size; ++j) {
              if (i == j || board[j] <= 0) continue;
              if (board[i] == board[j]) {
                return false;
              }
              if (std::abs(board[i] - board[j]) == std::abs(i - j)) return false;
            }
          }
        }
      
        return true;
      }
      
      bool Board::is_full() const {
        for (int i = 0; i < size; ++i) {
          if (board[i] <= 0) return false;
        }
      
        return true;
      }
      
      bool Board::has_symmetries() const {
        // Diese Funktion soll überprüfen, ob die Position Symmetrien aufweist.
        // Die beiden relevanten Symmetrien sind:
        // (A): Die Dame steht immer oben in der ersten Hälfte der Spalte.
        // Das Brett kann horizontal gespiegelt werden, wenn nötig.
        // (B): Die Dame in der letzten Spalte hat immer eine niedrigere Position als
        // die Dame in der ersten Spalte. Sonst kann das Schachbrett vertikal
        // gespiegelt werden, um dasselbe Ergebnis zu erzielen.
      
        // Wir gehen davon aus, dass das Brett nicht leer ist.
        assert(size > 0);
      
        // Wir prüfen zuerst die Symmetrie (A).
        // Dazu iterieren wir über die Spalten von links nach rechts und schauen, ob
        // die Dame in der oberen Hälfte steht.
        for (int i = 0; i < size; ++i) {
          // Wenn die Spalte leer ist, überspringen wir sie.
          if (board[i] <= 0) continue;
          // Wenn die Dame in der unteren Hälfte steht, gibt es keine Symmetrie (A).
          if (board[i] > size / 2) return false;
          // Wenn die Dame genau in der Mitte steht und die Größe ungerade ist, gibt
          // es keine Symmetrie (A).
          if (board[i] == size / 2 + 1 && size % 2 == 1) return false;
        }
      
        // Wir prüfen nun die Symmetrie (B).
        // Dazu vergleichen wir die Positionen der Damen in der ersten und letzten
        // Spalte.
        int first =
            -1;  // Die Position der Dame in der ersten Spalte oder -1, wenn leer.
        int last =
            -1;  // Die Position der Dame in der letzten Spalte oder -1, wenn leer.
      
        for (int i = 0; i < size; ++i) {
          // Wir suchen nach der ersten nicht-leeren Spalte von links.
          if (first == -1 && board[i] > 0) {
            first = board[i];
          }
          // Wir suchen nach der ersten nicht-leeren Spalte von rechts.
          if (last == -1 && board[size - i - 1] > 0) {
            last = board[size - i - 1];
          }
          // Wenn wir beide gefunden haben, brechen wir die Schleife ab.
          if (first != -1 && last != -1) break;
        }
      
        // Wenn beide Spalten leer sind, gibt es keine Symmetrie (B).
        if (first == -1 && last == -1) return false;
      
        // Wenn nur eine Spalte leer ist, gibt es keine Symmetrie (B).
        if (first == -1 || last == -1) return false;
      
        // Wenn die Dame in der letzten Spalte eine höhere Position hat als die Dame
        // in der ersten Spalte, gibt es keine Symmetrie (B).
        if (last > first) return false;
      
        // Ansonsten gibt es beide Symmetrien.
        return true;
      }
      
      void Board::horizontal_flip() {
        // Diese Funktion soll das Brett horizontal spiegeln.
      
        // Wir gehen davon aus, dass das Brett nicht leer ist und keine negativen
        // Werte enthält.
        assert(size > 0);
      
        for (int i = 0; i < size; ++i) {
          assert(board[i] >= 0);
          // Wenn die Spalte leer ist, überspringen wir sie.
          if (board[i] == 0) continue;
          // Wir spiegeln die Position der Dame um die Mitte des Bretts.
          board[i] = size + 1 - board[i];
      
          /* Alternativ könnte man auch einfach das Array umdrehen:
             int temp = board[i];
             board[i] = board[size - i - 1];
             board[size - i - 1] = temp;
             Aber das wäre weniger effizient, da man mehr Zuweisungen braucht. */
        }
      }
      
      void Board::vertical_flip() {
        // Diese Funktion soll das Brett vertikal spiegeln.
        // Wir gehen davon aus, dass das Brett nicht leer ist und keine negativen
        // Werte enthält.
        assert(size > 0);
        // Wir tauschen einfach die Spalten von links und rechts aus, bis wir die
        // Mitte erreichen.
        for (int i = 0; i < size / 2; ++i) {
          assert(board[i] >= 0);
          assert(board[size - i - 1] >= 0);
          // Wir suchen nach der ersten nicht-leeren Spalte von rechts.
          // Wir brauchen nicht zu prüfen, ob die letzte Spalte leer ist, sondern nur
          // ob die aktuelle Spalte von rechts nicht leer ist.
          if (board[size - i - 1] > 0) {  // Wir entfernen die Bedingung auf last
            // Wir speichern die Position der Dame in der linken Spalte in einer
            // temporären Variable.
            int temp = board[i];
            // Wir setzen die Position der Dame in der linken Spalte gleich der
            // Position der Dame in der rechten Spalte.
            board[i] = board[size - i - 1];
            // Wir setzen die Position der Dame in der rechten Spalte gleich der
            // temporären Variable.
            board[size - i - 1] = temp;
          }
        }
      }
      
      void Board::symmetry_normalize() {
        // Diese Funktion soll die Position spiegeln, falls nötig, um die fundamentale
        // Lösung zu erhalten.
      
        // Wir gehen davon aus, dass das Brett nicht leer ist und keine negativen
        // Werte enthält.
        assert(size > 0);
      
        // Wir prüfen zuerst die Symmetrie (A).
        // Dazu iterieren wir über die Spalten von links nach rechts und schauen, ob
        // die Dame in der oberen Hälfte steht.
        bool need_horizontal_flip =
            false;  // Diese Variable wird auf true gesetzt, wenn wir eine horizontale
                    // Spiegelung brauchen.
        for (int i = 0; i < size; ++i) {
          assert(board[i] >= 0);
          // Wenn die Spalte leer ist, überspringen wir sie.
          if (board[i] == 0) continue;
          // Wenn die Dame in der unteren Hälfte steht, brauchen wir eine horizontale
          // Spiegelung.
          if (board[i] > size / 2) {
            need_horizontal_flip = true;
            break;
          }
          // Wenn die Dame genau in der Mitte steht und die Größe ungerade ist,
          // brauchen wir auch eine horizontale Spiegelung.
          if (board[i] == size / 2 + 1 && size % 2 == 1) {
            need_horizontal_flip = true;
            break;
          }
        }
      
        // Wir führen die horizontale Spiegelung durch, falls nötig.
        if (need_horizontal_flip) {
          horizontal_flip();
        }
      
        // Wir prüfen nun die Symmetrie (B).
        // Dazu vergleichen wir die Positionen der Damen in der ersten und letzten
        // Spalte.
        int first =
            -1;  // Die Position der Dame in der ersten Spalte oder -1, wenn leer.
        int last =
            -1;  // Die Position der Dame in der letzten Spalte oder -1, wenn leer.
      
        for (int i = 0; i < size; ++i) {
          // Wir suchen nach der ersten nicht-leeren Spalte von links.
          if (first == -1 && board[i] > 0) {
            first = board[i];
          }
          // Wir suchen nach der ersten nicht-leeren Spalte von rechts.
          if (last == -1 && board[size - i - 1] > 0) {
            last = board[size - i - 1];
          }
          // Wenn wir beide gefunden haben, brechen wir die Schleife ab.
          if (first != -1 && last != -1) break;
        }
      
        // Wenn beide Spalten leer sind oder nur eine Spalte leer ist, gibt es keine
        // Symmetrie (B).
        if (first == -1 || last == -1) return;
      
        // Wenn die Dame in der letzten Spalte eine höhere Position hat als die Dame
        // in der ersten Spalte, brauchen wir eine vertikale Spiegelung.
        if (last > first) {
          vertical_flip();
        }
      }
      
      bool Board::operator<(const Board &n) const {
        // Diese Funktion soll eine Ordnung über Positionen definieren: kürzere
        // Bretter sind kürzer als lange Bretter. Wenn die Länge gleich ist, dann
        // vergleichen Bretter wie im Wörterbuch (lexikographische Ordnung) von links
        // nach rechts (kleinere Zahlen bedeuten kleiner, wenn gleich wird sich das
        // nächste Element angeschaut).
      
        // Wir vergleichen zuerst die Größe der Bretter.
        if (size < n.size) return true;
        if (size > n.size) return false;
      
        // Wenn die Größe gleich ist, dann vergleichen wir die Spalten von links nach
        // rechts.
        for (int i = 0; i < size; ++i) {
          // Wir vergleichen die Positionen der Damen in den Spalten.
          // Negative Werte werden als negative Zahlen behandelt.
          if (board[i] < n.board[i]) return true;
          if (board[i] > n.board[i]) return false;
          // Wenn die Positionen gleich sind, gehen wir zur nächsten Spalte.
        }
      
        // Wenn alle Spalten gleich sind, sind die Bretter gleich und somit nicht
        // kleiner.
        return false;
      }
      
      bool Board::operator<=(const Board &n) const {
        // Diese Funktion soll eine Ordnung über Positionen definieren: kürzere
        // Bretter sind kürzer als lange Bretter. Wenn die Länge gleich ist, dann

        // nach rechts (kleinere Zahlen bedeuten kleiner, wenn gleich wird sich das

        // Wir vergleichen zuerst die Größe der Bretter.
        if (size < n.size) return true;
        if (size > n.size) return false;

          if (board[i] > n.board[i]) return false;
          // Wenn die Positionen gleich sind, gehen wir zur nächsten Spalte.
        }
      
        // Wenn alle Spalten gleich sind, sind die Bretter gleich und somit kleiner
        // oder gleich.
        return true;
      }

      void NQueens::check_and_add_child(Node *parent, bool direction,
                                        std::stack &stack) {
        /*ÜB10, Aufgabe 6, optional*/
        Board b = parent->get_configuration();
        if (!b.has_next(direction)) {
          bdd.connect_false(parent, direction);
          return;
        }
      
        b.next(direction);
        const bool valid = b.is_valid();
        if (!valid) {
          bdd.connect_false(parent, direction);
          return;
        }
        if (b.is_full()) {
          assert(valid);
          bdd.connect_true(parent, direction);
          return;
        } else {
          bool already_exists = false;
          Node *child = bdd.import_node(b, already_exists);
          if (!already_exists) stack.push(child);
          bdd.connect(parent, child, direction);
        }
        return;
      }
      
      void NQueens::construct_bdd() {
        Board b(size);  // Wir erstellen ein leeres Brett mit der gegebenen Größe.
      
        std::stack stack;  // Wir erstellen einen Stack für die Positionen,
                                   // die noch zu bearbeiten sind.
      
        bool already_exists = false;
      
        Node *n =
            bdd.import_node(b, already_exists);  // Wir importieren das leere Brett
                                                 // als ersten Knoten im BDD
      
        stack.push(n);  // Wir fügen den ersten Knoten in den Stack ein.
      
        int next_limit = size;  // Wir setzen eine Grenze für die Anzahl der besuchten
                                // Knoten, bevor wir den BDD reduzieren.
      
        int visited_nodes = 1;  // Wir zählen die Anzahl der besuchten Knoten
      
        while (!stack.empty()) {  // Solange der Stack nicht leer ist, wiederholen wir
                                  // folgende Schritte:
      
          Node *n = stack.top();  // Wir nehmen den obersten Knoten aus dem Stack
      
          stack.pop();
      
          check_and_add_child(
              n, false, stack);  // Wir prüfen und fügen den Nachfolger hinzu, wenn
                                 // die Dame nicht an der nächsten Position ist
      
          check_and_add_child(n, true,
                              stack);  // Wir prüfen und fügen den Nachfolger hinzu,
                                       // wenn die Dame an der nächsten Position ist.
      
          visited_nodes += 2;  // Wir erhöhen die Anzahl der besuchten Knoten um zwei
                               // (für beide Nachfolger)
      
          if (visited_nodes >= next_limit) {  // Wenn wir die Grenze erreicht haben,
                                              // reduzieren wir den BDD
      
            next_limit *= 2;  // Wir verdoppeln die Grenze für das nächste Mal
      
            bdd.normalize_bdd();  // Wir rufen die Methode normalize_bdd auf, die alle
                                  // Pfade zu false entfernt.
          }
        }
      
        bdd.normalize_bdd();  // Am Ende reduzieren wir den BDD noch einmal.
      }

      template  void Stack::reserve(int n) {
        if (n < capacity)
          return;
      
        capacity = n;
        T *nb = new T[n];
        for (int i = 0; i < size; ++i)
          nb[i] = stack[i];
      
        delete[] stack;
        stack = nb;
      }
      
      template  void Stack::push(const T &board) {
        if (size + 1 >= capacity)
          reserve(2 * capacity);
      
        assert(size < capacity);
        stack[size] = std::move(board);
        ++size;
      }
      
      template  void Stack::pop() {
        if (!size)
          return;
        --size;
      }
      
      template  T Stack::top() const { return stack[size - 1]; }
      
      template  bool Stack::empty() const { return size == 0; }

      #ifndef BLATT_BDD_BDD
      #define BLATT_BDD_BDD
      
      #include 
      #include 
      #include 
      
      #include "node.hpp"
      
      // This is a very simple implementation of a BDD (binary decision diagram).
      //
      // There are two special nodes `true_node' and `false_node' that represent 1 and
      // 0 of the BDD.
      //
      // The interface is very simple in order to construct the BDD:
      //
      //  - it is possible to import a Board. This returns the node as imported by the
      //  BDD
      //
      //  - Nodes can be connected via `connect_true' and `connect_false' (for
      //  connection to true or
      //  - false) and `connect' to connect two nodes
      //
      //
      // Internally, all nodes are stored in the `nodes' vector in order do be able to
      // destray all the nodes.
      //
      // NEW: internally we use the to_delete flag in nodes, but the invariant that is true
      // (except within functions) is that it is always set to false.
      class BDD {
      public:
        BDD() : fresh_id(0), root(nullptr) {}
      
        BDD(const BDD &) = delete;
        BDD(BDD &&) = delete;
      
        // This function imports a Board and returns a node. The successors of the
        // nodes are set to the false node automatically.
        //
        // NEW ÜB10
        // If the node has symmetries, we check if the fundamental node is already
        // present and set the already_exists parameter.
        Node *import_node(const Board &b, bool &already_exists);
      
        // destructor
        ~BDD() {
          for (auto &n : nodes) {
            delete n;
          }
        }
      
        friend class Node;
      
        // connect two nodes together
        // overwriting any existing relationships.
        void connect(Node *parent, Node *children, bool if_true);
      
        // connect the node to the true or false node assuming that parent is valid
        // pointer to a node (neither true nor false)
        void connect_true(Node *parent, bool if_true);
        void connect_false(Node *parent, bool if_true);
      
        // counts the number of paths leading to true
        int count_true_path() const;
      
        // count the number of fundamental paths leading to true
        /*NEW ÜB 10, Aufgabe 12*/
        int count_fundamental_true_path() const;
      
        // Write the graph as a dot (graphviz) file.

      // Representation of a configuration for our n-queens
      // problem and iteration over all positions.
      //
      // Our class internally uses an array of int of a given size, where each int
      // means:
      //   - == 0 no queens in this column
      //   - <0 no queen up to that position
      //   - >0 the only queen is at that position.
      //
      // The board is already tailored toward our needs by disallowing 2 queens in
      // this column. We also are using a fixed size.
      //
      // For each board, there are at most two successors: either the queen is at the
      // next empty position (the column value was <= 0 and has now value > 0) or it
      // is not (the column value was <=0 and is now decreased by 1).
      class Board {
      public:
        Board() : size(0), board(nullptr) {}
      
        Board(int n) : size(n) {
          board = new int[n];
          for (int i = 0; i < n; ++i)
            board[i] = 0;
        }
      
        Board(const Board &pos) {
          size = pos.size;
          board = new int[size];
          for (int i = 0; i < size; ++i)
            board[i] = pos.board[i];
        }
      
        Board(Board &&pos) {
          size = pos.size;
          board = pos.board;
          pos.size = 0;
          pos.board = nullptr;
        }
      
        Board operator=(const Board &pos) {
          delete[] board;
          size = pos.size;
          board = new int[size];
          for (int i = 0; i < size; ++i)
            board[i] = pos.board[i];
          return *this;
        }
      
        Board operator=(Board &&pos) {
          size = pos.size;
          board = pos.board;
          pos.size = 0;
          pos.board = nullptr;
          return *this;
        }
      
        ~Board() { delete[] board; }
      
        bool operator==(const Board &b) const {
          if (size != b.size)
            return false;
          for (int i = 0; i < size; ++i) {
            if (board[i] != b.board[i])
              return false;
          }
          return true;
        }
      
        // checks if this position has a next position
        bool has_next(bool taken) const;
      
        // changes the current position to the next one
        // assuming that there is one
        void next(bool taken);
      
        // give a board as a string with '|' separating the cells with 'D' for queens
        // and when there is no queen, it prints '.' for positions where there is no
        // queen. For example |D|.|_|
        // |_|.|_|
        // |_| |_|
        // here the queen in the second column can only be at the very bottomx.
        friend std::ostream &operator<<(std::ostream &stream, const Board &board);
      
        // checks if the current board is compatible with the rules
        bool is_valid() const;
      
        // checks if the current board is compatible with the rules
        bool is_full() const;
      
        bool has_symmetries() const;
        void symmetry_normalize();
      
        // precondition: no negative numbers!
        void horizontal_flip();
        void vertical_flip();
      
        // Shortlex order on the elements on the board.
        // Negative entries are considered as negative numbers.
        /*NEW ÜB 10, Aufgabe 9*/
        bool operator<(const Board &n) const;
      
        // Shortlex order on the elements on the board
        // Negative entries are considered as negative numbers.
        /*NEW ÜB 10, Aufgabe 9*/
        bool operator<=(const Board &n) const;
      
      private:
        int size;
        int *board;
        FRIEND_TEST(BoardTest, nextTest);
        FRIEND_TEST(BoardTest, stringTest);
        FRIEND_TEST(BoardTest, validTest);
        FRIEND_TEST(BoardTest, flipTest);
